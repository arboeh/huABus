#include <tunables/global>

profile huawei_solar_modbus_mqtt flags=(attach_disconnected,mediate_deleted) {
  #include <abstractions/base>

  # Python interpreter
  /usr/bin/python3* ix,
  /usr/lib/python3*/** mr,
  
  # Add-on data & config
  /data/** rw,
  /config/** r,
  /share/** r,

  # Networking (Modbus TCP + MQTT)
  network inet stream,
  network inet6 stream,
  network inet dgram,
  network inet6 dgram,

  # Temp files
  /tmp/** rw,
  
  # Deny dangerous operations
  deny /proc/sys/kernel/** wl,
  deny /sys/kernel/security/** wl,
  deny @{PROC}/sys/vm/** wl,
}
